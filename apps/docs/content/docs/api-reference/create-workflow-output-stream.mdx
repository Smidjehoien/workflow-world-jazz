# `createWorkflowOutputStream()`

Returns a writable stream during a workflow run.

You should use this in your workflows to produce streaming output.

## Import

```typescript
import { createWorkflowOutputStream } from '@vercel/workflow-core/runtime';
```

## API Signature

### Parameters

The `createWorkflowOutputStream` function takes no parameters.

### Returns

The `createWorkflowOutputStream` function returns a type `WritableStream`.

<auto-type-table
  type="WritableStream"
/>

## Examples

Here's a simple example of how you can use `createWorkflowOutputStream` inside your workflow.

```typescript lineNumbers
import { sleep } from '@vercel/workflow-core';
import { createWorkflowOutputStream } from '@vercel/workflow-core/runtime'; // [!code highlight]

export async function outputStreamWorkflow() {
  "use workflow";
  
  const writable = createWorkflowOutputStream(); // [!code highlight]

  await sleep("1s")
  await stepWithOutputStreamObject(writable);
  await sleep("1s");
  await stepCloseOutputStream(writable);
  
  return 'done';
}

async function stepWithOutputStreamObject(writable: WritableStream) {
  "use step";

  const writer = writable.getWriter();
  await writer.write({ foo: "test" }); // Object data
  writer.releaseLock();
}

async function stepCloseOutputStream(writable: WritableStream) {
  "use step";

  await writable.close();
}

```
